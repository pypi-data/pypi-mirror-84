# python-qcmd
Synchronous concurrent executor command queueing for Python 3.8

# Developing
## Install pipx if pipenv is not installed
python3 -m pip install pipx
python3 -m pipx ensurepath

## Install pipenv using pipx
pipx install pipenv

## Init
bash init.sh
